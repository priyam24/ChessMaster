class rook{
	int m;
	int n;
	public:
		rook(){
		}
		rook(int pos1,int pos2){
			m=pos1;
			n=pos2;
		}
		int istrue(int k,int l){
			if(k==m&&l==n)
			return 1;
			else return 0;
		}
		int virtual_move(int,int,int,int,char,board);
		void final_move(int,int,int,int,char,board&);
		void vanish(rook &x){
			x.m=-1;
			x.n=-1;
		}
		int check(board,char);
};

int rook::virtual_move(int s1,int r1,int s,int r,char c,board br){
	int chk;
	char victim;
	victim=br.b[s][r];
	if(c=='b')
	br.b[s][r]='R';
	else br.b[s][r]='r';
	br.b[s1][r1]='-';
	if(s==m&&r==n)
	return 100;
	if(s1!=m||r1!=n)
	return 100;
	if((s>m&&r>n)||(s>m&&r<n)||(s<m&&r>n)||(s<m&&r<n))
	return 1;
	if(s>m){
		for(int i=m+1;i<s;i++){
			if(br.b[i][n]!='-')
			return 1;
		}
	}
	else if(s<m){
		for(int i=m-1;i>s;i--){
			if(br.b[i][n]!='-')
			return 1;
		}
	}
	else{
		if(r>n)
		for(int i=n+1;i<r;i++){
			if(br.b[m][i]!='-')
			return 1;
		}
		if(r<n)
		for(int i=n-1;i>r;i--){
			if(br.b[m][i]!='-')
			return 1;
	    }
	}
	if(c=='b'){
		if(victim!='-'){
			if(victim>=65&&victim<=90)
			return 1;
			else{
				switch(victim){
					case 'q':
						
						chk=check(br,'b');
						if(chk==0)
						return 2;
						else return 1;
					case 'k':
					
						chk=check(br,'b');
						if(chk==0)
						return 3;
						else return 1;
					case 'b':
					
						chk=check(br,'b');
						if(chk==0)
						return 5;
						else return 1;
					case 'r':
					
						chk=check(br,'b');
						if(chk==0)
						return 4;
						else return 1;
					case 'p':
					
						chk=check(br,'b');
						if(chk==0)
						return 6;
						else return 1;
					default:
						return 1;
				}
			}
		}
		else{
			chk=check(br,'b');
			if(chk==0)
			return 0;
			else return 1;	
		}
	}
	else if(c=='w'){
		if(victim!='-'){
			if(victim>=97&&victim<=122)
			return 1;
			else{
				switch(victim){
					case 'Q':
						
						chk=check(br,'w');
						if(chk==0)
						return 22;
						else return 1;
					case 'K':
						
						chk=check(br,'w');
						if(chk==0)
						return 33;
						else return 1;
					case 'B':
					
						chk=check(br,'w');
						if(chk==0)
						return 55;
						else return 1;
					case 'R':
				
						chk=check(br,'w');
						if(chk==0)
						return 44;
						else return 1;
					case 'P':
					
						chk=check(br,'w');
						if(chk==0)
						return 66;
					default:
						return 1;
				}
			}
		}
		else{
			chk=check(br,'w');
			if(chk==0)
			return 0;
			else return 1;	
		}
	}
